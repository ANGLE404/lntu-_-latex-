% 定义高级透明边框命令（带阴影、圆角、彩色边框）
\newcommand{\transparentbox}[1]{%
    \noindent\begin{tcolorbox}[
        enhanced,
        colback=white!0,  % 完全透明背景
        colframe=primary_color,  % 主色调边框
        boxrule=0.5pt,  % 边框粗细
        arc=6pt,  % 圆角半径
        outer arc=6pt,  % 外圆角
        drop shadow={
            shadow xshift=2pt,
            shadow yshift=-2pt,
            shadow blur=4pt,
            opacity=0.3,
            fill=gray!50
        },  % 阴影效果
        left=4pt,
        right=4pt,
        top=1pt,
        bottom=1pt,
        width=\linewidth,
        before skip=0pt,  % 取消上方间距
        after skip=0pt   % 取消下方间距
    ]
        #1
    \end{tcolorbox}%
}

% 定义加宽版高级透明边框命令（用于内容较多的部分）

        enhanced,
        colback=white!0,  % 完全透明背景
        colframe=secondary_color,  % 次色调边框
        boxrule=0.5pt,  % 边框粗细
        arc=6pt,  % 圆角半径
        outer arc=6pt,  % 外圆角
        drop shadow={
            shadow xshift=2pt,
            shadow yshift=-2pt,
            shadow blur=4pt,
            opacity=0.3,
            fill=gray!50
        },  % 阴影效果
        left=6pt,
        right=6pt,
        top=3pt,
        bottom=3pt,
        width=\dimexpr\linewidth+11em\relax,
        before skip=0pt,  % 取消上方间距
        after skip=0pt   % 取消下方间距
    ]
        #1
    \end{tcolorbox}%
}
